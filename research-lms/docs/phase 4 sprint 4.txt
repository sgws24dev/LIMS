```csharp
public decimal QuantityAvailable => QuantityOnHand - QuantityReserved;

public bool IsLowStock =>
    ReorderPoint > 0 && QuantityOnHand <= ReorderPoint;

public bool IsOutOfStock => QuantityOnHand <= 0;

public bool IsExpiringSoon =>
    ExpiryDate.HasValue &&
    ExpiryDate.Value <= DateOnly.FromDateTime(DateTime.UtcNow.AddDays(30));

    ExpiryDate.HasValue &&
    ExpiryDate.Value < DateOnly.FromDateTime(DateTime.UtcNow);
```

**Domain method — stock mutation (enforces invariants):**
```csharp
public void AdjustStock(decimal delta, string reason)
{
    if (QuantityOnHand + delta < 0)
        throw new InvalidOperationException(
            $"Cannot reduce stock below zero. " +
            $"Current: {QuantityOnHand}, Delta: {delta}");
    QuantityOnHand += delta;
    UpdatedAt = DateTime.UtcNow;
}

public void ReserveStock(decimal quantity)
{
    if (quantity > QuantityAvailable)
        throw new InvalidOperationException(
            $"Insufficient available stock. " +
            $"Available: {QuantityAvailable}, Requested: {quantity}");
    QuantityReserved += quantity;
    UpdatedAt = DateTime.UtcNow;
}

public void ReleaseReservation(decimal quantity)
{
    QuantityReserved = Math.Max(0, QuantityReserved - quantity);
    UpdatedAt = DateTime.UtcNow;
}
```

`RecordStockReceiptCommand` Handler:
```csharp
// 1. Load item
var item = await _repo.GetByIdAsync(cmd.ItemId, ct)
    ?? throw new NotFoundException("Item not found.");

// 2. Snapshot state before
decimal before = item.QuantityOnHand;

// 3. Mutate via domain method
item.AdjustStock(cmd.Quantity, "Receipt");
item.UnitCost = cmd.UnitCost;   // update last known cost

// 4. Create transaction record
var transaction = new StockTransaction
{
    TenantId         = item.TenantId,
    InventoryItemId  = item.Id,
    Type             = StockTransactionType.Receipt,
    Quantity         = cmd.Quantity,
    QuantityBefore   = before,
    QuantityAfter    = item.QuantityOnHand,
    UnitCost         = cmd.UnitCost,
    TotalCost        = cmd.Quantity * cmd.UnitCost,
    ReferenceType    = cmd.PurchaseOrderId.HasValue ? "PurchaseOrder" : null,
    ReferenceId      = cmd.PurchaseOrderId,
    Notes            = cmd.Notes,
    TransactedById   = cmd.TransactedById,
    TransactedByName = cmd.TransactedByName,
    TransactedAt     = DateTime.UtcNow
};

// 5. Save item + transaction
await _itemRepo.UpdateAsync(item, ct);
await _txRepo.AddAsync(transaction, ct);

// 6. Check if stock no longer low → publish LowStockResolvedEvent
// (no event if it was already above reorder point)
if (before <= item.ReorderPoint && item.QuantityOnHand > item.ReorderPoint)
    await _bus.Publish(new LowStockResolvedEvent(item.Id, item.TenantId, item.Name));

return transaction.Id;
```

`AdjustStockCommand` Handler:
1. Load item
2. Compute delta = `cmd.NewQuantity - item.QuantityOnHand`
3. Call `item.AdjustStock(delta, cmd.Reason)` — throws if negative result
4. Create `StockTransaction` with `Type = Adjustment`, `Quantity = delta`
5. Save item + transaction

`WriteOffStockCommand` Handler:
1. Load item
2. Call `item.AdjustStock(-cmd.Quantity, cmd.Reason)` — throws if quantity
   exceeds current stock
3. Create transaction with `Type = Write_Off`, `Quantity = -cmd.Quantity`
4. Save

```csharp
public class CreateInventoryItemValidator
    : AbstractValidator<CreateInventoryItemCommand>

    public CreateInventoryItemValidator()
    {
        RuleFor(x => x.SKU).NotEmpty().MaximumLength(100)
            .Matches(@"^[A-Z0-9\-_]+$")
            .WithMessage("SKU must contain only uppercase letters, numbers, hyphens, and underscores.");
        RuleFor(x => x.Name).NotEmpty().MaximumLength(300);
        RuleFor(x => x.Description).MaximumLength(2000);
        RuleFor(x => x.ReorderPoint).GreaterThanOrEqualTo(0);
        RuleFor(x => x.ReorderQuantity).GreaterThan(0);
```

```csharp
public bool CanTransitionTo(PurchaseOrderStatus newStatus) =>
    (Status, newStatus) switch
    {
        (PurchaseOrderStatus.Draft,              PurchaseOrderStatus.PendingApproval)   => true,
        (PurchaseOrderStatus.Draft,              PurchaseOrderStatus.Cancelled)         => true,
        (PurchaseOrderStatus.PendingApproval,    PurchaseOrderStatus.Approved)          => true,
        (PurchaseOrderStatus.PendingApproval,    PurchaseOrderStatus.Draft)             => true, // reject back
        (PurchaseOrderStatus.PendingApproval,    PurchaseOrderStatus.Cancelled)         => true,
        (PurchaseOrderStatus.Approved,           PurchaseOrderStatus.Ordered)           => true,

        (PurchaseOrderStatus.Ordered,            PurchaseOrderStatus.PartiallyReceived) => true,

        (PurchaseOrderStatus.PartiallyReceived,  PurchaseOrderStatus.FullyReceived)     => true,
        (PurchaseOrderStatus.PartiallyReceived,  PurchaseOrderStatus.Cancelled)         => true,
```

```csharp
// This is the most complex handler — full implementation:
var po = await _poRepo.GetByIdWithLinesAsync(cmd.PurchaseOrderId, ct)
    ?? throw new NotFoundException("Purchase order not found.");

if (po.Status != PurchaseOrderStatus.Ordered &&
    po.Status != PurchaseOrderStatus.PartiallyReceived)
    throw new InvalidOperationException(
        "Can only receive items on Ordered or PartiallyReceived purchase orders.");

foreach (var receipt in cmd.ReceivedLines)
{
    var line = po.Lines.FirstOrDefault(l => l.Id == receipt.LineId)
        ?? throw new NotFoundException($"PO line {receipt.LineId} not found.");

    if (receipt.QuantityReceived <= 0)
        continue;

    if (line.QuantityReceived + receipt.QuantityReceived > line.QuantityOrdered)
        throw new InvalidOperationException(
            $"Received quantity for '{line.Description ?? "item"}' " +
            $"exceeds ordered quantity.");

    // Update line received qty
    line.QuantityReceived += receipt.QuantityReceived;

    // Record stock receipt transaction
    await _mediator.Send(new RecordStockReceiptCommand(
        ItemId:          line.InventoryItemId,
        Quantity:        receipt.QuantityReceived,
        UnitCost:        line.UnitPrice,
        PurchaseOrderId: po.Id,
        Notes:           $"Received via PO {po.PONumber}",
        TransactedById:  cmd.ReceivedById,
        TransactedByName: cmd.ReceivedByName
    ), ct);
}

// Update PO status based on line completion
bool allReceived = po.Lines.All(l => l.IsFullyReceived);
bool anyReceived = po.Lines.Any(l => l.QuantityReceived > 0);

po.Status = allReceived
    ? PurchaseOrderStatus.FullyReceived
    : anyReceived
        ? PurchaseOrderStatus.PartiallyReceived
        : po.Status;

if (allReceived)
{
    po.ReceivedAt = DateTime.UtcNow;
    // Update vendor stats
    var vendor = await _vendorRepo.GetByIdAsync(po.VendorId, ct);
    if (vendor is not null)
    {
        vendor.TotalOrdersCount++;
        vendor.TotalOrdersValue += po.TotalAmount;
        await _vendorRepo.UpdateAsync(vendor, ct);
    }
    // Publish event
    await _bus.Publish(new PurchaseOrderFullyReceivedEvent(
        po.Id, po.TenantId, po.PONumber, po.VendorId));
}

await _poRepo.UpdateAsync(po, ct);
```

```csharp
public class LowStockCheckJob(
    IInventoryItemRepository itemRepo,
    IPublishEndpoint bus,
    ILogger<LowStockCheckJob> logger) : ILowStockCheckJob
{
    public async Task ExecuteAsync()
    {
        var lowStockItems = await itemRepo.GetLowStockItemsAsync(
            CancellationToken.None);

        foreach (var item in lowStockItems)
        {
            try
            {
                await bus.Publish(new LowStockAlertEvent(
                    item.Id,
                    item.TenantId,
                    item.Name,
                    item.SKU,
                    item.QuantityOnHand,
                    item.ReorderPoint));

                logger.LogInformation(
                    "Low stock alert published for Item {SKU} " +
                    "(OnHand: {Qty}, ReorderPoint: {Rp})",
                    item.SKU, item.QuantityOnHand, item.ReorderPoint);
            }
            catch (Exception ex)
            {
                logger.LogError(ex,
                    "Failed to publish low-stock alert for {ItemId}", item.Id);
            }
        }
    }
}
```

Register in DI:
```csharp
recurringJobManager.AddOrUpdate<ILowStockCheckJob>(
    "low-stock-check",
    job => job.ExecuteAsync(),
    "0 * * * *"   // Every hour
);
```

```csharp
public static IServiceCollection AddInfrastructureServices(

        });
```

Create `frontend/src/modules/inventory/components/barcode-scanner.tsx`:
```tsx
import { Html5QrcodeScanner } from 'html5-qrcode';

interface BarcodeScannerProps {
  onScan: (barcode: string) => void;
  onClose: () => void;
}

export const BarcodeScanner: React.FC<BarcodeScannerProps> = ({
  onScan, onClose
}) => {
  const scannerRef = useRef<Html5QrcodeScanner | null>(null);

  useEffect(() => {
    scannerRef.current = new Html5QrcodeScanner(
      "barcode-reader",
      { fps: 10, qrbox: { width: 250, height: 250 } },
      false
    );

    scannerRef.current.render(
      (decodedText) => {
        onScan(decodedText);
        scannerRef.current?.clear();
      },
      (error) => { /* Ignore scan errors */ }
    );

    return () => { scannerRef.current?.clear(); };
  }, []);

  return (
    <Dialog open onOpenChange={onClose}>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Scan Barcode / QR Code</DialogTitle>
        </DialogHeader>
        <div id="barcode-reader" className="w-full" />
        <p className="text-sm text-muted-foreground text-center mt-2">
          Point your camera at the barcode or QR code
        </p>
      </DialogContent>
    </Dialog>
  );
};
```

**After scan:**
1. Close scanner modal
2. Call `GET /api/v1/inventory/items/by-barcode/{barcode}`
3. If found → navigate to `/inventory/items/{id}`
4. If not found → show toast "No item found for this barcode. Create one?" +
   "Create Item" button pre-filled with barcode value
